def djelitelj(broj,djelitelj):
    return broj%djelitelj==0
        
def funkcija(a):
    brojac=0
    for i in range(a,0,-1):
       if djelitelj(a,i):
           brojac+=1
    return brojac
    
x = int(input("Unesite broj: "))
while x <= 0:
    x = int(input("Unesite pozitivan broj: "))

print("Broj djelitelja: ",  funkcija(x))
