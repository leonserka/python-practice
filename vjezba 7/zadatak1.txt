import random

def funkcija(p):
    x=random.randint(1, 100)
    if p == "par":
        while x%2!=0:
             x=random.randint(1, 100)
        return x

    elif p=="nepar":
        while x%2==0:
             x=random.randint(1, 100)
        return x

y=input("Unesite par ili nepar: ")
print("Broj: ", funkcija(y))
